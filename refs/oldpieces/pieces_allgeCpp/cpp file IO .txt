#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main () {
   string line;
   string line_from_output;
   bool isequal=true;
   int linenum=0;
   ifstream standerdfile ("..\\data\\standerd_output.txt");
   ifstream myfile ("..\\data\\output.txt");
   while ( getline (myfile,line_from_output) )
   {
         getline (standerdfile,line);
         linenum++;
         if(line_from_output.compare(line)!=0){
            isequal=false;
            break;
         }
   }
   ofstream report ("..\\data\\result.txt");
   if(isequal)
      report<<"they are equal txts\n";
   else 
      report<<"not equal, at line:"<< linenum <<'\n';

   report.close();
   standerdfile.close();
   myfile.close();
   return 0;
}
